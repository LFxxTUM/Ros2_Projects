% ---------- Code listings ----------
\usepackage{listings}
\usepackage{textcomp}   % gerade Anfuehrungszeichen (fuer upquote)
\usepackage{etoolbox}   % Hooks, um Listings als Block zusammenzuhalten
% Jedes Listing in eine minipage -> wird nicht ueber einen Seitenumbruch
% zerrissen (Listings, die laenger als eine Seite sind, vorher aufteilen).
\BeforeBeginEnvironment{lstlisting}{\par\noindent\begin{minipage}{\linewidth}}
\AfterEndEnvironment{lstlisting}{\end{minipage}\par\medskip}
\lstdefinestyle{base}{
  backgroundcolor=\color{codebg},
  basicstyle=\ttfamily\footnotesize,
  keywordstyle=\color{codekw}\bfseries,
  commentstyle=\color{codecom}\itshape,
  stringstyle=\color{codestr},
  numbers=none,           % keine Zeilennummern -> sauberes Kopieren aus dem PDF
  showstringspaces=false, breaklines=true, frame=single,
  rulecolor=\color{rule}, framesep=5pt, xleftmargin=6pt,
  tabsize=2, captionpos=b, aboveskip=8pt, belowskip=6pt,
  % --- saubere Kopierbarkeit aus dem PDF ---
  columns=fullflexible,   % keine eingefuegten Extra-Leerzeichen
  keepspaces=true,        % Originalabstaende erhalten
  upquote=true,           % ' und ` als gerade Zeichen (Shell-Kommandos!)
}
\lstdefinestyle{py}{style=base,language=Python}
\lstdefinestyle{cpp}{style=base,language=C++}
\lstdefinestyle{bash}{style=base,language=bash,keywordstyle=\color{accent2}\bfseries}
\lstdefinestyle{xml}{style=base,language=XML,%
  morekeywords={link,joint,visual,collision,inertial,origin,axis,parent,child,%
    limit,mass,inertia,geometry,mesh,robot,transmission,actuator}}
\lstdefinestyle{yaml}{style=base,basicstyle=\ttfamily\footnotesize}

% ---------- Screenshot placeholder ----------
% Marks exactly where to paste a real screenshot. Replace the tcolorbox line
% with \includegraphics once the PNG sits in figures/.
% Usage: \screenshotplaceholder[width-frac]{file-basename}{caption / what to capture}
\newcommand{\screenshotplaceholder}[3][0.8]{%
  \begin{figure}[htbp]\centering
  \begin{tcolorbox}[enhanced,width=#1\linewidth,colback=gray!5,colframe=gray!50,
      boxrule=0pt,arc=3pt,halign=center,valign=center,height=3.0cm,
      borderline={0.8pt}{0pt}{gray!55,dashed}]
    {\bfseries\color{gray!65}[\,Screenshot einf\"ugen\,]}\\[3pt]
    {\ttfamily\scriptsize figures/#2.png}\\[2pt]
    {\scriptsize ersetze diese Box durch
      \texttt{\textbackslash includegraphics[width=#1\textbackslash textwidth]\{#2\}}}
  \end{tcolorbox}
  \caption{#3}\label{fig:#2}
  \end{figure}}
